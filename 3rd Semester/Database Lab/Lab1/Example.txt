CREATE TABLE DEPARTMENT(
DepartmentName VARCHAR2(4),
Establishment DATE,
Budget NUMBER,
Location VARCHAR2(20),
DepartmentID NUMBER,
CONSTRAINT PK_DEPARTMENT PRIMARY KEY (DepartmentName)
);

CREATE TABLE STUDENT(
SID NUMBER,
StudentName VARCHAR2(20),
DepartmentName VARCHAR2(4),
BirthDate DATE,
CGPA NUMBER,
CONSTRAINT PK_STUDENT PRIMARY KEY (SID, StudentName),
CONSTRAINT FK_StdDeptName FOREIGN KEY (DepartmentName) REFERENCES DEPARTMENT(DepartmentName)
);

CREATE TABLE EMPLOYEE(
EmployeeName VARCHAR2(20),
EmployeeID NUMBER,
DepartmentName VARCHAR2(4),
Salary NUMBER,
Joining DATE,
CONSTRAINT PK_Employee PRIMARY KEY (EmployeeID, EmployeeName),
CONSTRAINT FK_EmpDeptName FOREIGN KEY (DepartmentName) REFERENCES DEPARTMENT(DepartmentName)
);
