// 1.5
SELECT Pieces.Name
FROM Pieces, Provides
WHERE Pieces.Code = Provides.Piece AND Provider = 'HAL';

// 1.6
SELECT Pieces.Name, Provides.Provider, Nested.MAX_PRICE
FROM
(
	SELECT Provides.Piece, MAX(Provides.Price) MAX_PRICE
	FROM Provides
	GROUP BY Provides.Piece
) Nested, Pieces, Provides
WHERE Nested.Piece = Provides.Piece AND Nested.Piece = Pieces.Code;
