//Creating tables
CREATE TABLE DEPARTMENT(
Name VARCHAR2(5),
ID Number,
Est Number,
CONSTRAINT PK_DeptID PRIMARY KEY (ID)
);

CREATE TABLE STUDENTS(
Name VARCHAR2(20),
ID Number,
DoB number,
CGPA Number,
DepartmentID number,
CONSTRAINT PK_StdID PRIMARY KEY (ID),
CONSTRAINT FK_DeptID FOREIGN KEY (DepartmentID) REFERENCES DEPARTMENT(ID) 
);
